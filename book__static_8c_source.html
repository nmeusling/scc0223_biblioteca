176  * memory that was allocated to the book is freed. Return 0 if book removed, 0

179 int remove_book_booklist(book_list *books, int prev_book){

193  return 0;
194  }
195  //book is last but not only element
196  else if(books->elements[prev_book].next == books->last)
197  {
198  books->last = prev_book;

202  remove_wait_list(&books->elements[index], books);
203  free_node_booklist(books, index);
204  return 0;
